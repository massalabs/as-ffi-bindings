30impl Read<Vec<u8>> for BufferPtr {
31    fn read(&self, memory: &Memory, store: &impl AsStoreRef) -> anyhow::Result<Vec<u8>> {
32        let size = self.size(memory, store)?;
33        let memory_view = memory.view(store);
34        let wasm_slice_ = self.0.slice(&memory_view, size);
35        if let Ok(wasm_slice) = wasm_slice_ {
36            let mut res = vec![0; size as usize];
37            wasm_slice.read_slice(&mut res)?;
38            Ok(res)
39        } else {
40            anyhow::bail!("Wrong offset: can't read buf")
41        }
42    }
43
44    fn size(&self, memory: &Memory, store: &impl AsStoreRef) -> anyhow::Result<u32> {
45        let memory_view = memory.view(&store);
46        let u32_size = size_of::<u32>() as u32;
47        let ptr = self.0.sub_offset(u32_size)?;
48        let slice_len_buf = ptr.slice(&memory_view, u32_size)?.read_to_vec()?;
49        Ok(u32::from_ne_bytes(slice_len_buf.try_into().map_err(
50            |v| anyhow::Error::msg(format!("Unable to convert vec: {:?} to &[u8; 4]", v)),
51        )?))
52    }
53}
